stack* push(stack* top)
{
    stack* newnode;
    int newinfo;
    printf("Enterthe new information to be push on stacktop\n");
    scanf("%d",&newinfo);
    newnode=(stack*)malloc(sizeof(stack));
    newnode->info=newinfo;
    if (top == NULL) 
    {
        top=newnode;
        top->next=NULL;
    } 
    else
    { 
        newnode->next=top;
        top=newnode;
    } 
    return(top);
}
stack* pop(stack* top)
{
    int item; 
    stack *temp; 
    if( top == NULL) 
    {
        printf(" Empty Stack …"); 
        return(NULL) ; 
    }
    else
    { 
        temp=top;
        top=top->next; 
        temp->next=NULL; 
        free(temp); 
    }
    
    return(top); 
}
void display(stack *top) 
{
    stack *temp;
    temp = top;
    if(top == NULL) 
    {
       printf("Empty stack …");
       exit(0);
    }
    else 
    {
        printf("Top->");
        while(temp != NULL) 
        {
            printf("%d ->", temp->info);
            temp = temp->next;
        }
    }
}
